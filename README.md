import turtle as t

t.speed(30)
t.pensize(20)

t.circle(100)

t.up()
t.forward(240)
t.down()
t.pencolor('red')
t.circle(100)

t.up()
t.backward(480)
t.down()
t.pencolor('blue')
t.circle(100)

t.up()
t.forward(240)
t.down()
t.pencolor('black')
t.circle(100)

t.up()
t.backward(120)
t.goto(-110,-100)
t.down()
t.pencolor('yellow')
t.circle(100)

t.up()
t.forward(240)
t.goto(120,-100)
t.down()
t.pencolor('green')
t.circle(100)






t.done()
